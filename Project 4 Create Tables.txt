Project 4 Create Tables

CREATE TABLE Student (
    studentID Integer GENERATED BY DEFAULT ON NULL AS IDENTITY,
    fullName varchar(100),
    Address varchar(100),
    email varchar(100),
    gpa number,
    ssn integer,
    CONSTRAINT student_pk PRIMARY KEY (studentID));

CREATE TABLE Faculty (
    facultyID INTEGER GENERATED BY DEFAULT ON NULL AS IDENTITY,
    fullName varchar(100),
    email varchar(100),
    buildingName varchar(100),
    officeNumber integer,
    phoneNumber number(10),
    department varchar(100),
    positions varchar(100),
    CONSTRAINT faculty_pk PRIMARY KEY (facultyID)
);

CREATE TABLE Semester (
    semesterID INTEGER GENERATED BY DEFAULT ON NULL AS IDENTITY,
    periods varchar(100),
    years integer,
    Constraint semesterID_pk PRIMARY KEY (semesterID)
);

CREATE TABLE Course (
    courseID Integer GENERATED BY DEFAULT ON NULL AS IDENTITY,
    prefixs varchar(100),
    courseName varchar(100),
    days varchar(100),
    startTime varchar(100),
    endTime varchar(100),
    creditHours integer,
    CONSTRAINT courseID_pk PRIMARY KEY (courseID)
);
    

CREATE TABLE Schedule (
    scheduleID integer GENERATED BY DEFAULT ON NULL AS IDENTITY,
    courseID integer,
    facultyID integer,
    semesterID integer,
    CONSTRAINT schedule_pk PRIMARY KEY (scheduleID),
    CONSTRAINT schedule1_fk FOREIGN KEY (courseID) REFERENCES Course(courseID),
    CONSTRAINT schedule2_fk FOREIGN KEY (facultyID) REFERENCES Faculty(facultyID),
    CONSTRAINT schedule3_fk FOREIGN KEY (semesterID) REFERENCES Semester(semesterID));

CREATE TABLE Enrollment (
    enrollmentID integer GENERATED BY DEFAULT ON NULL AS IDENTITY,
    courseID integer,
    studentID integer,
    semesterID integer,
    CONSTRAINT enrollment_pk PRIMARY KEY (enrollmentID),
    CONSTRAINT enrollment1_fk FOREIGN KEY (courseID) REFERENCES Course(courseID),
    CONSTRAINT enrollment2_fk FOREIGN KEY (studentID) REFERENCES Student(studentID),
    CONSTRAINT enrollment3_fk FOREIGN KEY (semesterID) REFERENCES Semester(semesterID));
